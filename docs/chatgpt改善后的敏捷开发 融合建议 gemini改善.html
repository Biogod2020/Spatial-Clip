import torch
import torch.nn as nn

class SpatialMultimodalModel(nn.Module):
    def __init__(self, image_encoder, gene_encoder, fusion_module, task_head):
        super().__init__()
        self.image_encoder = image_encoder
        self.gene_encoder = gene_encoder
        self.fusion_module = fusion_module
        self.task_head = task_head

    def forward(self, image_patch, gene_vector):
        # Input shapes:
        # image_patch: [B, 3, 224, 224]
        # gene_vector: [B, 2000] # Assuming 2000 HVGs

        # 1. Encode each modality
        # z_img: [B, 512]
        z_img = self.image_encoder(image_patch)

        # z_gene: [B, 256]
        z_gene = self.gene_encoder(gene_vector)

        # 2. Fuse representations
        # z_fused: [B, 768] or whatever the fusion output is
        z_fused = self.fusion_module(z_img, z_gene)

        # 3. Pass through task-specific head
        # logits: [B, num_classes]
        logits = self.task_head(z_fused)

        return logits
